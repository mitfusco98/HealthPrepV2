@app.route('/api/submit-signup', methods=['POST'])
@limiter.limit("3 per hour")  # Strict rate limit for signups
def submit_signup():
    try:
        data = request.get_json()
        
        # Validate and sanitize all input data
        validated_data = validate_signup_data(data)
        
        print(f"Forwarding signup request to Health Prep API for: {validated_data.get('organization_name')}")
        
        # Forward validated data to main app
        response = requests.post(
            HEALTH_PREP_API_URL,  # https://health-prep-v-201-mitchfusillo.replit.app/api/signup
            json=validated_data,
            headers={'Content-Type': 'application/json'},
            timeout=30
        )
        
        if response.status_code == 200:
            api_response = response.json()
            organization_id = api_response.get('organization_id', 'Unknown')
            
            # Send notification email
            send_signup_notification(validated_data, organization_id)
            
            return jsonify(api_response), 200
        else:
            # Extract error message from Health Prep API response
            error_message = 'Failed to create organization'
            try:
                error_data = response.json()
                error_message = error_data.get('error', error_message)
            except:
                error_message = response.text or error_message
            
            print(f"Error from Health Prep API: {error_message}")
            return jsonify({
                'success': False,
                'error': error_message
            }), response.status_code
